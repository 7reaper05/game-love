function unlock() {

  const pass = document.getElementById("code").value;

  const music = document.getElementById("bgMusic");


  if(pass === "yourpasscode") {

    lockScreen.style.display = "none";

    mainContent.style.display = "block";


    music.volume = 0;

    music.play();


    let vol = 0;

    const fade = setInterval(() => {

      if (vol < 0.6) {

        vol += 0.05;

        music.volume = vol;

      } else clearInterval(fade);

    }, 200);

  }

}

/* 🌌 BLACK HOLE */

const canvas = document.getElementById("blackHoleCanvas");

const ctx = canvas.getContext("2d");


canvas.width = window.innerWidth;

canvas.height = window.innerHeight;


let particles = [];


for (let i = 0; i < 800; i++) {

  particles.push({

    angle: Math.random() * Math.PI * 2,

    radius: Math.random() * 300,

    speed: 0.002 + Math.random() * 0.004

  });

}


function animate() {

  ctx.fillStyle = "rgba(0,0,0,0.2)";

  ctx.fillRect(0,0,canvas.width,canvas.height);


  particles.forEach(p => {

    p.angle += p.speed;

    p.radius *= 0.995;


    if (p.radius < 2) p.radius = 300;


    let x = canvas.width/2 + Math.cos(p.angle)*p.radius;

    let y = canvas.height/2 + Math.sin(p.angle)*p.radius;


    ctx.beginPath();

    ctx.arc(x,y,1.2,0,Math.PI*2);

    ctx.fillStyle="white";

    ctx.fill();

  });


  /* glow core */

  ctx.beginPath();

  ctx.arc(canvas.width/2,canvas.height/2,40,0,Math.PI*2);

  ctx.fillStyle="rgba(255,255,255,0.05)";

  ctx.fill();


  requestAnimationFrame(animate);

}

animate();
